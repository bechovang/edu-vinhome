# 1. Cài PostgreSQL
choco install postgresql

# 2. Khởi động dịch vụ
net start postgresql-x64

# 3. Tạo database
psql -U postgres -c "DROP DATABASE IF EXISTS student_management;"
psql -U postgres -c "CREATE DATABASE student_management WITH ENCODING 'UTF8' TEMPLATE template0 OWNER postgres;"

# 4. Tạo bảng
psql -U postgres -d student_management -f "C:\Users\Admin\Desktop\GIT CLONE\edu-vinhome\database\create_tables.sql"

# 5. Chèn dữ liệu
psql -U postgres -d student_management -f "C:\Users\Admin\Desktop\GIT CLONE\edu-vinhome\database\insert_data.sql"

# 6. Kiểm tra
psql -U postgres -d student_management -c "SELECT * FROM students;"
